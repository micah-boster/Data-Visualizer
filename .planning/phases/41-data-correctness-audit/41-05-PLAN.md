// src/lib/computation/penetration-rate.smoke.ts
/**
 * DCR-03 scope-rollup invariant for dollar-weighted penetration rate.
 *
 * Asserts that the same value emerges at three scope levels:
 *   1. KPI card     (computeKpis on all rows of a (partner, product) pair)
 *   2. Summary row  (buildPairSummaryRows partner-row cell)
 *   3. Direct math  (Σ(rate_i × placed_i) / Σ placed_i over the pair's batches)
 *
 * If any of the three diverge, partner-level blending or aggregation
 * regression has crept in. Pinned by Plan 41-01 (DCR-01) + Plan 41-05.
 */

import assert from 'node:assert/strict';
import { readFileSync } from 'node:fs';
import { resolve } from 'node:path';

import { computeKpis } from './compute-kpis.ts';
import { buildPairSummaryRows } from '../columns/root-columns.ts';

// Load fixture (a real multi-partner cached payload)
const FIXTURE_PATH = resolve(import.meta.dirname, '../static-cache/batch-summary.json');
const allRows = JSON.parse(readFileSync(FIXTURE_PATH, 'utf-8')) as Record<string, unknown>[];

// Pick a (partner, product) pair with multiple batches and varying placed sizes
const pair = (() => {
  const seen = new Map<string, Record<string, unknown>[]>();
  for (const row of allRows) {
    const key = `${row.PARTNER_NAME}::${row.ACCOUNT_TYPE}`;
    const existing = seen.get(key) ?? [];
    existing.push(row);
    seen.set(key, existing);
  }
  // Find the first pair with ≥ 3 batches and varied TOTAL_AMOUNT_PLACED
  for (const [key, rows] of seen.entries()) {
    if (rows.length < 3) continue;
    const placedValues = new Set(rows.map((r) => r.TOTAL_AMOUNT_PLACED));
    if (placedValues.size < 2) continue;
    return { key, rows };
  }
  throw new Error('No suitable (partner, product) pair in fixture');
})();

const TOLERANCE = 0.01; // 1 basis point on a 0..100 scale

// (1) Direct math
const sumWeighted = pair.rows.reduce((s, r) => {
  const v = Number(r.PENETRATION_RATE_POSSIBLE_AND_CONFIRMED);
  const w = Number(r.TOTAL_AMOUNT_PLACED);
  return Number.isFinite(v) && Number.isFinite(w) && w > 0 ? s + v * w : s;
}, 0);
const totalWeight = pair.rows.reduce((s, r) => {
  const w = Number(r.TOTAL_AMOUNT_PLACED);
  return Number.isFinite(w) && w > 0 ? s + w : s;
}, 0);
const directDollarWeighted = totalWeight > 0 ? sumWeighted / totalWeight : 0;

// (2) Summary row from buildPairSummaryRows
const summaryRows = buildPairSummaryRows(pair.rows);
const pairSummary = summaryRows.find((r) =>
  `${r.PARTNER_NAME}::${r.ACCOUNT_TYPE}` === pair.key,
);
assert(pairSummary, `summary row missing for ${pair.key}`);
const summaryValue = Number(pairSummary.PENETRATION_RATE_POSSIBLE_AND_CONFIRMED);

// (3) KPI card via computeKpis
const kpis = computeKpis(pair.rows);
// computeKpis exposes penetration via... [verify the actual KpiAggregates field
// shape: penetrationRate? penetrationRatePercent? document the actual field name
// in this comment]. Adapt to the actual shape:
const kpiPen = (kpis as unknown as { penetrationRate?: number }).penetrationRate
  ?? (kpis as unknown as { penetrationRatePossibleAndConfirmed?: number }).penetrationRatePossibleAndConfirmed
  ?? null;
assert(kpiPen !== null, 'computeKpis did not expose penetration rate field');

// Three-way invariant
assert.ok(
  Math.abs(directDollarWeighted - summaryValue) < TOLERANCE,
  `direct (${directDollarWeighted}) !== summary (${summaryValue}) for ${pair.key}`,
);
assert.ok(
  Math.abs(directDollarWeighted - kpiPen) < TOLERANCE,
  `direct (${directDollarWeighted}) !== kpi (${kpiPen}) for ${pair.key}`,
);

// Negative test: confirm arithmetic-mean would NOT match (catches DCR-01 regression)
const arithmeticMean = pair.rows.reduce((s, r) => {

/**
 * DCR-04 — Phase 39 (partner, product) scope-rollup audit.
 *
 * For every multi-product partner in the fixture, verify:
 *   1. Each (partner, product) summary row computes from rows whose
 *      ACCOUNT_TYPE matches the row's product (no cross-product blending)
 *   2. The pair-level KPI values match what computeKpis returns when fed
 *      ONLY the matching-product rows
 *   3. Summing per-pair totals across products of one partner equals the
 *      partner-as-whole values when the table is grouped by PARTNER_NAME
 *      (this is the apples-and-oranges check at scope-rollup grain — it's
 *      what would have been the partner-level value pre-Phase-39)
 *
 * Findings (DCR-04 status) feed into docs/METRIC-AUDIT.md.
 */

import assert from 'node:assert/strict';
import { readFileSync } from 'node:fs';
import { resolve } from 'node:path';

import { computeKpis } from './compute-kpis.ts';
import { buildPairSummaryRows } from '../columns/root-columns.ts';

const FIXTURE_PATH = resolve(import.meta.dirname, '../static-cache/batch-summary.json');
const allRows = JSON.parse(readFileSync(FIXTURE_PATH, 'utf-8')) as Record<string, unknown>[];

// Group rows by partner; identify multi-product partners
const byPartner = new Map<string, Record<string, unknown>[]>();
for (const row of allRows) {
  const partner = String(row.PARTNER_NAME);
  const existing = byPartner.get(partner) ?? [];
  existing.push(row);
  byPartner.set(partner, existing);
}

const multiProductPartners = [...byPartner.entries()].filter(([_, rows]) => {
  const products = new Set(rows.map((r) => String(r.ACCOUNT_TYPE)));
  return products.size > 1;
});

if (multiProductPartners.length === 0) {
  console.warn('scope-rollup smoke: no multi-product partner in fixture — skipping multi-product invariants');
}

const TOLERANCE = 0.01;

for (const [partner, partnerRows] of multiProductPartners) {
  // Build pair-summary rows for this partner only
  const summaryRows = buildPairSummaryRows(partnerRows);

  // Group partnerRows by product
  const byProduct = new Map<string, Record<string, unknown>[]>();
  for (const row of partnerRows) {
    const prod = String(row.ACCOUNT_TYPE);
    const arr = byProduct.get(prod) ?? [];
    arr.push(row);
    byProduct.set(prod, arr);
  }

  // INVARIANT 1: pair-summary uses ONLY matching-product rows
  for (const summary of summaryRows) {
    const product = String(summary.ACCOUNT_TYPE);
    const matchingRows = byProduct.get(product) ?? [];
    const otherProducts = [...byProduct.keys()].filter((p) => p !== product);

    // Sum of TOTAL_AMOUNT_PLACED across matching rows ONLY
    const expectedPlaced = matchingRows.reduce(
      (s, r) => s + (Number(r.TOTAL_AMOUNT_PLACED) || 0),
      0,
    );
    const summaryPlaced = Number(summary.TOTAL_AMOUNT_PLACED);

    assert.ok(
      Math.abs(expectedPlaced - summaryPlaced) < TOLERANCE,
      `[DCR-04] cross-product blending detected for ${partner} (${product}): summary placed=${summaryPlaced} but matching-only=${expectedPlaced}. Other products: ${otherProducts.join(', ')}`,
    );
  }

  // INVARIANT 2: pair-level KPI on matching rows === pair-level KPI from buildPairSummaryRows numbers
  for (const [product, rows] of byProduct.entries()) {
    const kpis = computeKpis(rows);
    const summary = summaryRows.find((s) => String(s.ACCOUNT_TYPE) === product);
    assert(summary, `summary row missing for ${partner}::${product}`);

    // Pick a stable scalar — totalCollected
    const summaryCollected = Number(summary.TOTAL_COLLECTED_LIFE_TIME);
    const kpiCollected = (kpis as unknown as { totalCollected?: number }).totalCollected;
    assert(kpiCollected !== undefined, 'computeKpis did not expose totalCollected');
    assert.ok(
      Math.abs(summaryCollected - kpiCollected) < TOLERANCE,
      `[DCR-04] ${partner}::${product}: summary totalCollected=${summaryCollected} !== computeKpis(matching rows).totalCollected=${kpiCollected}`,
    );
  }

  // INVARIANT 3: sum of per-product totals === partner-as-whole when computed apples-and-oranges
  // (This is the legitimate partner-level number — sum across non-overlapping products.)
  const sumOfPerProductCollected = [...byProduct.values()].reduce(
    (s, rows) => s + computeKpis(rows).totalCollected,
    0,
  );
  const partnerWideCollected = computeKpis(partnerRows).totalCollected;
  assert.ok(
    Math.abs(sumOfPerProductCollected - partnerWideCollected) < TOLERANCE,
    `[DCR-04 / DCR-10] ${partner}: sum-of-per-product totalCollected=${sumOfPerProductCollected} !== partner-wide=${partnerWideCollected}`,
  );
}

console.log(`scope-rollup smoke OK (multi-product partners checked: ${multiProductPartners.length})`);
