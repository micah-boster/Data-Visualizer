    .filter(p => {
      const key = pairKey(p);
      if (seen.has(key)) return false;
      seen.add(key);
      return true;

// Compute revenueModelsPerPair: how many distinct revenue models exist for each (partner, product)

    if (p.revenueModel) set.add(p.revenueModel);
    m.set(k, set);

  return m;
