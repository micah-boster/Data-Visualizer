**File B: `src/components/partner-lists/attribute-filter-bar.tsx`**

A minimal additive row of attribute multi-selects. v1 renders ONLY ACCOUNT_TYPE (CONTEXT 2026-04-18 lock — PRODUCT_TYPE and REVENUE_BAND deferred). Structure should make adding more attributes later a one-line change.

Structure:
```tsx
'use client';

import { FilterCombobox } from '@/components/filters/filter-combobox';
import type { PartnerListFilters } from '@/lib/partner-lists/types';

const ATTRIBUTES = [
  {
    key: 'ACCOUNT_TYPE' as const,
    label: 'Account Type',
  },
  // Future: add { key: 'PRODUCT_TYPE', label: 'Product Type' }, etc. Rendering is data-driven.
] satisfies ReadonlyArray<{ key: keyof PartnerListFilters; label: string }>;

export function AttributeFilterBar({ availableValues, value, onChange }: AttributeFilterBarProps) {
  return (
    <div className="flex flex-wrap gap-inline">
      {ATTRIBUTES.map((attr) => {
        const values = availableValues[attr.key] ?? [];
        if (values.length === 0) return null;  // hide the combobox if no values exist (e.g. column missing from current dataset)

        return (
          <FilterCombobox
            key={attr.key}
            label={attr.label}
            options={values}
            selected={value[attr.key] ?? []}
            onChange={(next) => onChange({ ...value, [attr.key]: next })}
            placeholder={`Filter by ${attr.label}`}
          />
        );
      })}
    </div>
  );
}
```

**Internal state:**
```tsx
const { lists, createList, getList, renameList, hasListWithName /* , updateList */ } = usePartnerLists();
const [name, setName] = useState('');
const [filters, setFilters] = useState<PartnerListFilters>({});
const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
const [nameError, setNameError] = useState<string | null>(null);
```

**Derived state (useMemo):**

1. `deduplicatedPartners: { id: string; name: string }[]` — dedup `allRows` by `getPartnerName(row)`. `id` and `name` are the same partner name string (matches the Phase 25 convention).

2. `availableValues: Partial<Record<'ACCOUNT_TYPE', string[]>>` — collect unique non-null `row.ACCOUNT_TYPE` values from `allRows`, sorted alphabetically. If the column has zero values (edge case), `availableValues.ACCOUNT_TYPE` is an empty array; AttributeFilterBar will skip rendering the combobox.

3. `filteredPartnerIds: Set<string>` — when `filters.ACCOUNT_TYPE?.length > 0`, call `evaluateFilters(allRows, filters)` to get the set of matching partner names. When no filter is set, this is `null` (meaning "all partners available"). Use `useMemo([allRows, filters])`.

4. `availablePartners` — `deduplicatedPartners.filter(p => !selectedIds.has(p.id) && (!filteredPartnerIds || filteredPartnerIds.has(p.id)))`.

5. `selectedPartners` — `deduplicatedPartners.filter(p => selectedIds.has(p.id))`. (Selected pane always shows fully selected partners regardless of attribute filter — attribute filter only scopes Available.)

6. `canSave` — `name.trim().length > 0 && selectedIds.size > 0`.

**Edit-mode hydration (useEffect on editMode.listId change):**
```tsx
useEffect(() => {
  if (!editMode) {
    // Reset to fresh-create state
    setName('');
    setFilters({});
    setSelectedIds(new Set());
    setNameError(null);
    return;
  }
  const existing = getList(editMode.listId);
  if (!existing) {
    onOpenChange(false);
    toast('List not found', { description: 'It may have been deleted.' });
    return;
  }
  setName(existing.name);
  setFilters(existing.filters);
  setSelectedIds(new Set(existing.partnerIds));
}, [editMode, getList, onOpenChange]);
```

Reset also when `open` transitions from true to false.

**Save handler:**
```tsx
const handleSave = () => {
  const trimmed = name.trim();
  if (!trimmed) { setNameError('Name is required'); return; }
  // Name-collision check — only for create, or for edit-rename to a different-but-taken name
  if (!editMode && hasListWithName(trimmed)) {
    setNameError('A list with this name already exists');
    return;
  }
  if (editMode) {
    const existing = getList(editMode.listId);
    if (existing && existing.name !== trimmed && hasListWithName(trimmed)) {
      setNameError('A list with this name already exists');
      return;
    }
  }

  const partnerIds = Array.from(selectedIds);
  const hasAnyFilter = Object.values(filters).some((arr) => arr && arr.length > 0);
  const source: 'attribute' | 'manual' = hasAnyFilter ? 'attribute' : 'manual';

  if (editMode) {
    // Update existing list — name, filters, partnerIds all mutable in edit mode.
    // Plan 01 must ship `updateList(id, { name, partnerIds, filters })`. If it didn't, extend the
    // hook now: add a single-parameter bag updater that also bumps updatedAt.
    updateList(editMode.listId, { name: trimmed, partnerIds, filters });
    toast('List updated', { description: `"${trimmed}" saved` });
  } else {
    createList({ name: trimmed, partnerIds, filters, source });
    toast('List created', { description: `"${trimmed}" is ready to use` });
  }

  onOpenChange(false);
};
```
