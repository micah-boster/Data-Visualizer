<objective>
Add the optional segment-split dimension to charts and KPIs. Users activate it via a per-view toggle that only appears when the active pair has segments configured. Split state is independent per view (chart toggle and KPI toggle are separate). Pair-level rolled-up stays the default.

Purpose: PCFG-07 is the payoff for the full Phase 39 investment — once selection is pair-aware (Plan 01) and segments exist (Plan 02), this plan exposes them in the analytic surfaces. Users can now answer "how is Snap's Spanish-speaking cohort performing relative to English?" with a one-click toggle, once the underlying column exists in the data.

Output: Segment-split toggle on Collection Curve chart + KPI card block, "Segment" entry in Chart Builder's series dropdown (gated on pair config), and a shared compute helper that guarantees segment-level totals reconcile with pair-level totals (apples-and-oranges invariant at segment granularity).
</objective>

    **A. segment-split.ts**
    ```typescript
    import { evaluateSegments } from './segment-evaluator';
    import type { SegmentRule } from './types';
    import { computeKpis } from '@/lib/computation/compute-kpis';
    import { reshapeCurves } from '@/lib/computation/reshape-curves';
    import type { KpiAggregates } from '@/types/kpis';      // or correct path — grep to confirm
    import type { CurveSeries } from '@/types/curves';      // or correct path

    /**
     * Virtual column name used by GenericChart to add a synthetic "segment"
     * axis to the series-pivot pipeline. Keeps the chart schema stable: no new
     * literal variant, just a sentinel column string.
     */
    export const SEGMENT_VIRTUAL_COLUMN = '__SEGMENT__';

    /**
     * Split pair-scoped rows into per-segment buckets + Other.
     * Returns a stable ordering: segments in their configured order, followed
     * by 'Other' last.
     */
    export function splitRowsBySegment(
      rows: Array<Record<string, unknown>>,
      segments: SegmentRule[],
    ): Array<{ label: string; rows: Array<Record<string, unknown>>; isOther: boolean }> {
      const { bySegment, other } = evaluateSegments(rows, segments);
      const buckets = segments.map((s) => ({
        label: s.name,
        rows: bySegment.get(s.name) ?? [],
        isOther: false,
      }));
      if (other.length > 0) {
        buckets.push({ label: 'Other', rows: other, isOther: true });
      }
      return buckets;
    }

    /**
     * Compute KPI aggregates per segment plus an Other bucket.
     * Sum of segment + Other totalCollected MUST equal the pair-rollup
     * totalCollected (apples-and-oranges invariant). The smoke test asserts
     * this.
     */
    export function kpiAggregatesPerSegment(
      rows: Array<Record<string, unknown>>,
      segments: SegmentRule[],
    ): Array<{ label: string; kpis: KpiAggregates; isOther: boolean }> {
      return splitRowsBySegment(rows, segments).map(({ label, rows: segRows, isOther }) => ({
        label,
        kpis: computeKpis(segRows),
        isOther,
      }));
    }

    /**
     * Reshape curves per segment. Each entry carries the segment label and the
     * curve series array for that segment's rows.
     */
    export function reshapeCurvesPerSegment(
      rows: Array<Record<string, unknown>>,
      segments: SegmentRule[],
    ): Array<{ label: string; curves: CurveSeries[]; isOther: boolean }> {
      return splitRowsBySegment(rows, segments).map(({ label, rows: segRows, isOther }) => ({
        label,
        curves: reshapeCurves(segRows),
        isOther,
      }));
    }

    /**
     * Tag each row with its segment name (or 'Other'). Used by GenericChart's
     * row-prep pipeline to feed the series pivot — mutating the row avoids
     * copying large datasets. Caller must pass a cloned array if preservation
     * is needed.
     */
    export function tagRowsWithSegment(
      rows: Array<Record<string, unknown>>,
      segments: SegmentRule[],
    ): Array<Record<string, unknown>> {
      const { bySegment, other } = evaluateSegments(rows, segments);
      const rowToLabel = new Map<Record<string, unknown>, string>();
      for (const s of segments) {
        for (const r of bySegment.get(s.name) ?? []) rowToLabel.set(r, s.name);
      }
      for (const r of other) rowToLabel.set(r, 'Other');
      return rows.map((r) => ({ ...r, [SEGMENT_VIRTUAL_COLUMN]: rowToLabel.get(r) ?? 'Other' }));
    }
    ```
