```tsx
'use client';
import { Switch } from '@/components/ui/switch';
import { Label } from '@/components/ui/label';
import { usePartnerNorms } from '@/contexts/partner-norms';

export function HeatmapToggle() {
  const { heatmapEnabled, toggleHeatmap, norms } = usePartnerNorms();
  // Only render when norms are available (partner drill-down level)
  if (!norms) return null;
  return (
    <div className="flex items-center gap-1.5">
      <Switch
        id="heatmap-toggle"
        checked={heatmapEnabled}
        onCheckedChange={toggleHeatmap}
        className="scale-75"
      />
      <Label htmlFor="heatmap-toggle" className="text-xs cursor-pointer">
        Heatmap
      </Label>
    </div>
  );
}
```

Add deviation formatting path between the existing negative check (#2) and the existing outlier/threshold check (#3). The logic:

1. Import `usePartnerNorms` from context, `computeDeviation`, `formatDeviationTooltip`, `HEATMAP_COLUMNS` from formatting, and `getFormatter` for formatting the mean value.
2. Inside `FormattedCell`, call `usePartnerNorms()` to get `{ norms, heatmapEnabled }`.
3. After the negative check (step 2) and before the existing threshold check (step 3), add a NEW step 2.5:

```tsx
// 2.5 Deviation check — partner-level heatmap (only for eligible columns)
if (
  typeof value === 'number' &&
  norms &&
  heatmapEnabled &&
  HEATMAP_COLUMNS.has(columnKey)
) {
  const norm = norms[columnKey];
  if (norm) {
    const deviation = computeDeviation(value, norm);
    if (deviation && deviation.direction !== 'neutral') {
      // Use inline style for dynamic opacity (Tailwind can't do runtime values)
      const bgColor = deviation.direction === 'above'
        ? `oklch(0.55 0.15 145 / ${deviation.opacity})`   // green
        : `oklch(0.55 0.15 25 / ${deviation.opacity})`;    // red
      const darkBgColor = deviation.direction === 'above'
        ? `oklch(0.45 0.12 145 / ${deviation.opacity})`
        : `oklch(0.45 0.12 25 / ${deviation.opacity})`;

      // Format the mean value using the same formatter as the cell value
      const formatter = getFormatter(type);
      const formattedMean = formatter(norm.mean);
      const tooltipText = formatDeviationTooltip(
        formattedValue,
        formattedMean,
        deviation.percentDeviation,
      );

      return (
        <Tooltip>
          <TooltipTrigger
            className="rounded px-1 -mx-1 transition-colors duration-150"
            style={{ backgroundColor: bgColor }}
          >
            {formattedValue}
          </TooltipTrigger>
          <TooltipContent>{tooltipText}</TooltipContent>
        </Tooltip>
      );
    }
  }
}
```

**IMPORTANT:** For dark mode support, detect dark mode via a CSS approach. The simplest method: use CSS custom properties. Add two new CSS variables `--deviation-green` and `--deviation-red` to globals.css (in Task 2 below) with oklch values, then reference them in the inline style. Alternatively, since the opacity is dynamic, use inline styles with the oklch values directly — the oklch hue (145 for green, 25 for red) works in both light and dark because the opacity controls brightness perception. Use a single color per direction with the dynamic opacity.

**CRITICAL:** The existing threshold check (step 3) MUST remain unchanged and only fire when the deviation check does NOT apply. This preserves root-level static formatting. The natural flow handles this: if the deviation check matched and returned early, the threshold check never runs. If norms is null (root level) or heatmapEnabled is false or column not in HEATMAP_COLUMNS, the deviation check is skipped and threshold check runs as before.

**NOTE on `getFormatter`:** It's already imported in formatted-cell.tsx for getCellRenderer. For the FormattedCell component itself, add import of `getFormatter` if not already available in its scope.
