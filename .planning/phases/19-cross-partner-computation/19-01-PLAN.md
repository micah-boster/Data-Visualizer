**3. Create `src/lib/computation/compute-cross-partner.ts`:**

This module does all cross-partner computation in a single pass:

```typescript
import { quantileRank } from 'simple-statistics';
import type {
  KpiAggregates,
  BatchCurve,
  CurvePoint,
  PercentileRanks,
  AverageCurve,
  CrossPartnerEntry,
  CrossPartnerData,
  PartnerActivityStatus,
} from '@/types/partner-stats';
import { computeKpis } from './compute-kpis';
import { reshapeCurves } from './reshape-curves';
import { COLLECTION_MONTHS } from './reshape-curves';
```

**Core functions to implement:**

a) `groupByPartner(allRows)` -- Groups rows by PARTNER_NAME into a Map. Same pattern as `computeAllPartnerAnomalies`.

b) `classifyActivity(rows)` -- Determines partner activity status:
   - Look at the newest batch: find the row with the maximum BATCH_AGE_IN_MONTHS value (which is actually days). Wait -- BATCH_AGE_IN_MONTHS is in days per the reshape-curves code (`Math.floor(rawAge / 30)`). The SMALLEST value of BATCH_AGE_IN_MONTHS means the most recent batch.
   - Find `minAge = Math.min(...rows.map(r => Number(r.BATCH_AGE_IN_MONTHS) || Infinity))`.
   - Convert to months: `newestBatchMonths = Math.floor(minAge / 30)`.
   - If newestBatchMonths > 12*30... wait. BATCH_AGE_IN_MONTHS stores DAYS. So 6 months = ~180 days, 12 months = ~365 days.
   - `if (minAge > 365) return 'inactive'` (12+ months no new batches)
   - `if (minAge > 180) return 'semi-inactive'` (6-12 months)
   - Otherwise `return 'active'`

c) `computeAverageCurve(curves: BatchCurve[], mode: 'equal' | 'dollar-weighted')` -- Computes average recovery rate at each COLLECTION_MONTH:
   - For each month in COLLECTION_MONTHS:
     - Filter curves that have a point at that month
     - If fewer than 2 contributing curves, STOP (truncation rule)
     - If mode === 'equal': average the recoveryRate values
     - If mode === 'dollar-weighted': weighted average using totalPlaced as weight
     - Push resulting CurvePoint (amount=0 since it's an average, recoveryRate=computed)
   - Return the CurvePoint array

d) `computePercentileRanks(entries: CrossPartnerEntry[])` -- For each of the 5 metrics, compute percentile rank using `quantileRank`:
   - Filter to rank-eligible entries only
   - For each metric, collect values into an array, then for each partner compute `quantileRank(values, partnerValue)`
   - IMPORTANT: `quantileRank` from simple-statistics takes a SORTED array. Sort the values array before passing.
   - Set `percentileRanks` on each eligible entry
   - For ineligible entries, leave `percentileRanks` as null

e) `detectPercentileOutliers(entries: CrossPartnerEntry[])` -- For each partner with percentileRanks, check if ANY rank is below 0.10 (10th percentile):
   - If so, set `isPercentileOutlier = true` and add the metric key to `outlierMetrics`

f) `computeCrossPartnerData(allRows: Record<string, unknown>[])` -- Main function, returns CrossPartnerData:
   1. Group by partner
   2. For each partner: compute KPIs via `computeKpis(rows)`, compute `perDollarPlacedRate = kpis.totalPlaced > 0 ? kpis.totalCollected / kpis.totalPlaced : 0`, reshape curves via `reshapeCurves(rows)`, compute average curve (both modes), classify activity, determine rank eligibility (`batchCount >= 3 && activityStatus !== 'inactive'`)
   3. Build CrossPartnerEntry for each partner
   4. Filter to rankedPartners (isRankEligible === true)
   5. Compute percentile ranks across ranked partners
   6. Detect percentile outliers
   7. Compute portfolio-wide average curve: combine ALL curves from eligible partners, compute average in both modes
   8. Return CrossPartnerData with partners Map, rankedPartners array, portfolioAverageCurve

Export: `computeCrossPartnerData`

import { useMemo } from 'react';

'use client';

import { createContext, useContext, useMemo, type ReactNode } from 'react';
import type { CrossPartnerData } from '@/types/partner-stats';
import { useAllPartnerStats } from '@/hooks/use-all-partner-stats';

interface CrossPartnerContextValue {
  /** Cross-partner computation data. Null if no data. */
  crossPartnerData: CrossPartnerData | null;
}

const CrossPartnerContext = createContext<CrossPartnerContextValue | null>(null);

interface CrossPartnerProviderProps {
  allRows: Record<string, unknown>[];
  children: ReactNode;
}

/**
 * Provides cross-partner stats to the component tree.
 * Nest inside AnomalyProvider (or alongside it) in DataDisplay.
 */
export function CrossPartnerProvider({ allRows, children }: CrossPartnerProviderProps) {
  const crossPartnerData = useAllPartnerStats(allRows);

  const value = useMemo<CrossPartnerContextValue>(
    () => ({ crossPartnerData }),
    [crossPartnerData],
  );

  return (
    <CrossPartnerContext.Provider value={value}>
      {children}
    </CrossPartnerContext.Provider>
  );
}

/**
 * Access cross-partner data from the nearest CrossPartnerProvider.
 * Must be used within a CrossPartnerProvider.
 */
export function useCrossPartnerContext(): CrossPartnerContextValue {
  const context = useContext(CrossPartnerContext);
  if (!context) {
    throw new Error('useCrossPartnerContext must be used within a CrossPartnerProvider');
  }
  return context;
}

const enrichedAnomalies = useMemo(() => {
  if (!crossPartnerData) return partnerAnomalies;

  // Clone the map and enrich with percentile outlier flags
  const enriched = new Map(partnerAnomalies);
  for (const [name, entry] of crossPartnerData.partners) {
    const existing = enriched.get(name);
    if (existing) {
      // Merge percentile flags into existing anomaly
      enriched.set(name, {
        ...existing,
        isPercentileOutlier: entry.isPercentileOutlier,
        percentileOutlierMetrics: entry.outlierMetrics.map(String),
      });
    } else {
      // Partner had no z-score anomaly but may be a percentile outlier
      enriched.set(name, {
        isFlagged: false,
        severityScore: 0,
        flaggedBatchCount: 0,
        totalBatchCount: entry.batchCount,
        latestBatch: null,
        batches: [],
        usedPortfolioFallback: false,
        isPercentileOutlier: entry.isPercentileOutlier,
        percentileOutlierMetrics: entry.outlierMetrics.map(String),
      });
    }
  }
  return enriched;
}, [partnerAnomalies, crossPartnerData]);
