const timeoutRef = useRef<ReturnType<typeof setTimeout>>();

const handleSend = useCallback((text: string) => {
  if (!text.trim()) return;

  // Clear previous messages for search-bar pattern (not chat)
  setMessages([]);

  // Clear any existing timeout
  if (timeoutRef.current) clearTimeout(timeoutRef.current);

  // Set 30s timeout
  timeoutRef.current = setTimeout(() => {
    stop();
  }, 30_000);

  setLastQuery(text);
  sendMessage({ text });
  setInputValue('');
}, [sendMessage, stop, setMessages]);

useEffect(() => {
  if (status === 'ready' || status === 'error') {
    if (timeoutRef.current) clearTimeout(timeoutRef.current);
  }
}, [status]);

// Cleanup on unmount
useEffect(() => {
  return () => {
    if (timeoutRef.current) clearTimeout(timeoutRef.current);

}, []);

```typescript
function QuerySearchBarWithContext({
  drillState,
  allData,
  onRemoveScope,
}: {
  drillState: DrillState;
  allData: Record<string, unknown>[];
  onRemoveScope: () => void;
}) {
  const { allPartnerAnomalies } = useAnomalyContext();

  const dataContext = useMemo(() => {
    // Build partner summaries from allData
    const partnerGroups = new Map<string, Record<string, unknown>[]>();
    for (const row of allData) {
      const name = String(row.PARTNER_NAME ?? '');
      if (!partnerGroups.has(name)) partnerGroups.set(name, []);
      partnerGroups.get(name)!.push(row);
    }

    const partners: PartnerSummary[] = Array.from(partnerGroups.entries()).map(
      ([name, rows]) => ({
        name,
        batchCount: rows.length,
        stats: {
          totalBatches: rows.length,
          totalAccounts: rows.reduce((sum, r) => sum + (Number(r.TOTAL_ACCOUNTS) || 0), 0),
          weightedPenetrationRate: 0, // simplified — exact computation not critical for prompt context
          collectionRate6mo: 0,
          collectionRate12mo: 0,
          totalCollected: rows.reduce((sum, r) => sum + (Number(r.TOTAL_COLLECTED) || 0), 0),
          totalPlaced: rows.reduce((sum, r) => sum + (Number(r.TOTAL_PLACED) || 0), 0),
        },
      }),
    );

    return buildDataContext(
      { level: drillState.level, partnerId: drillState.partner, batchId: drillState.batch },
      { partners, anomalies: allPartnerAnomalies },
    );
  }, [allData, drillState, allPartnerAnomalies]);

  return (
    <QuerySearchBar
      drillState={drillState}
      dataContext={dataContext}
      onRemoveScope={onRemoveScope}
    />
  );
}
```
