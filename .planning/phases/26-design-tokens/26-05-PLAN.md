**C. `src/components/tokens/token-card.tsx`** — reusable token display card:

```tsx
'use client';
import { Copy, Check } from 'lucide-react';
import { useState } from 'react';
import { toast } from 'sonner';

type TokenCardProps = {
  label: string;         // human name, e.g. "Card padding"
  cssVar: string;        // '--spacing-card-padding'
  tailwindClass?: string;// 'p-card-padding'
  value?: string;        // resolved value for display, e.g. '16px'
  children: React.ReactNode; // the live example
};

export function TokenCard({ label, cssVar, tailwindClass, value, children }: TokenCardProps) {
  const [copied, setCopied] = useState<'var' | 'class' | null>(null);

  async function copy(kind: 'var' | 'class', text: string) {
    await navigator.clipboard.writeText(text);
    setCopied(kind);
    toast.success('Copied');
    setTimeout(() => setCopied(null), 1500);
  }

  return (
    <div className="bg-surface-raised rounded-lg p-card-padding shadow-xs flex flex-col gap-stack">
      {/* Live example zone */}
      <div className="flex items-center justify-center p-section bg-surface-inset rounded-md min-h-[80px]">
        {children}
      </div>
      {/* Metadata row */}
      <div className="flex flex-col gap-[var(--spacing-1)]">
        <div className="text-label uppercase text-muted-foreground">{label}</div>
        <div className="flex items-center gap-inline">
          <code className="text-body-numeric text-foreground flex-1">{cssVar}{value ? ` → ${value}` : ''}</code>
          <button
            onClick={() => copy('var', `var(${cssVar})`)}
            aria-label="Copy CSS variable"
            className="focus-visible:ring-2 focus-visible:ring-ring rounded-sm p-[var(--spacing-1)]"
          >
            {copied === 'var' ? <Check className="size-3.5" /> : <Copy className="size-3.5" />}
          </button>
        </div>
        {tailwindClass && (
          <div className="flex items-center gap-inline">
            <code className="text-caption text-muted-foreground flex-1">{tailwindClass}</code>
            <button
              onClick={() => copy('class', tailwindClass)}
              aria-label="Copy Tailwind class"
              className="focus-visible:ring-2 focus-visible:ring-ring rounded-sm p-[var(--spacing-1)]"
            >
              {copied === 'class' ? <Check className="size-3.5" /> : <Copy className="size-3.5" />}
            </button>
          </div>
        )}
      </div>
    </div>
  );
}
```
