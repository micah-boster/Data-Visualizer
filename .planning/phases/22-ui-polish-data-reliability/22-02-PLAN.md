  Modify the existing fallback.ts to normalize empty strings to null and integrate schema validation.

  1. **Add normalizeRow and normalizeData functions** at the top of the file (after imports):
     ```typescript
     function normalizeRow(row: Record<string, unknown>): Record<string, unknown> {
       const normalized: Record<string, unknown> = {};
       for (const [key, value] of Object.entries(row)) {
         normalized[key] = value === "" ? null : value;
       }
       return normalized;
     }

     function normalizeData(response: DataResponse): DataResponse {
       return {
         ...response,
         data: response.data.map(normalizeRow),
       };
     }
     ```
