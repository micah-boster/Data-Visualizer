'use client';

import { Button } from '@/components/ui/button';
import * as DialogPrimitive from '@base-ui/react/dialog';
import { Save, Keyboard } from 'lucide-react';
import { useState } from 'react';

export function A11ySpecimen(): JSX.Element {
  const [pressed, setPressed] = useState(false);
  const [dialogOpen, setDialogOpen] = useState(false);

  return (
    <div className="flex flex-col gap-section p-card-padding">
      {/* Spec 1: focus-glow on a Button */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Focus ring — .focus-glow</h3>
        <p className="text-body text-muted-foreground">
          Tab-focusing any interactive element in the app reveals the soft glow (Phase 31 DS-31).
        </p>
        <div className="flex gap-inline">
          <Button variant="default">Default</Button>
          <Button variant="outline">Outline</Button>
          <Button variant="ghost">Ghost</Button>
        </div>
      </section>

      {/* Spec 2: Icon-only button labeling recipe */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Icon-only button with aria-label</h3>
        <p className="text-body text-muted-foreground">
          Tooltips alone don't satisfy axe-core button-name. Pair aria-label with Tooltip for sighted users.
        </p>
        <Button size="icon" aria-label="Save current view">
          <Save className="h-4 w-4" />
        </Button>
      </section>

      {/* Spec 3: aria-pressed toggle */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Toggle button — aria-pressed</h3>
        <p className="text-body text-muted-foreground">
          Toggle buttons expose aria-pressed reflecting state.
        </p>
        <Button
          variant="outline"
          aria-pressed={pressed}
          onClick={() => setPressed((v) => !v)}
        >
          Toggle me — {pressed ? 'on' : 'off'}
        </Button>
      </section>

      {/* Spec 4: Base UI Dialog with modal=true (focus trap + Escape + restore) */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Modal Dialog — focus trap (Base UI modal={`{true}`})</h3>
        <p className="text-body text-muted-foreground">
          Open the dialog: focus traps inside; Escape closes; focus returns to the trigger.
        </p>
        <DialogPrimitive.Root open={dialogOpen} onOpenChange={setDialogOpen} modal>
          <DialogPrimitive.Trigger render={<Button variant="outline">Open dialog</Button>} />
          <DialogPrimitive.Portal>
            <DialogPrimitive.Backdrop className="fixed inset-0 bg-black/30" />
            <DialogPrimitive.Popup className="surface-overlay fixed left-1/2 top-1/2 -translate-x-1/2 -translate-y-1/2 p-card-padding rounded-md">
              <DialogPrimitive.Title className="text-heading mb-stack">Trapped dialog</DialogPrimitive.Title>
              <p className="text-body mb-stack">Tab cycles within. Escape closes.</p>
              <DialogPrimitive.Close render={<Button variant="default">Close</Button>} />
            </DialogPrimitive.Popup>
          </DialogPrimitive.Portal>
        </DialogPrimitive.Root>
      </section>

      {/* Spec 5: sr-only skip-to-content link */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Skip-to-content link — sr-only + focus:not-sr-only</h3>
        <p className="text-body text-muted-foreground">
          Tab-focus reveals the normally-hidden link. Live in src/app/layout.tsx.
        </p>
        <a href="#__demo" className="sr-only focus:not-sr-only focus:inline-block focus-glow bg-surface-raised px-inline py-inline rounded-md text-body">
          Skip to content (Tab here)
        </a>
      </section>

      {/* Spec 6: Keyboard-flow row recipe */}
      <section className="flex flex-col gap-stack">
        <h3 className="text-heading">Row-level keyboard — Tab / Enter / Escape</h3>
        <p className="text-body text-muted-foreground">
          Drill-capable rows in the data table are Tab stops; Enter drills in; Escape returns.
        </p>
        <div role="table" className="surface-raised rounded-md">
          <div role="row" tabIndex={0} className="focus-glow-within flex items-center gap-inline px-card-padding py-stack rounded-md">
            <Keyboard className="h-4 w-4" aria-hidden="true" />
            <span className="text-body">Focusable row — Tab to focus, Enter to activate</span>
          </div>
        </div>
      </section>
    </div>
  );
}
