```ts
import { test, expect } from '@playwright/test';
import { setTheme } from './helpers/theme';

test('keyboard flow: root → partner drill → escape back to root', async ({ page }) => {
  await setTheme(page, 'light');
  await page.goto('/');
  await page.waitForLoadState('networkidle');
  await page.waitForSelector('table tbody tr');

  // Tab into the first drill-capable table row.
  // Toolbar + sidebar + table header absorb an unpredictable number of Tab stops.
  // Use keyboard-focus on the first <tr tabIndex=0> directly for determinism.
  const firstDrillableRow = page.locator('table tbody tr[tabindex="0"]').first();
  await firstDrillableRow.focus();
  await page.keyboard.press('Enter');

  // After Enter, drill URL should include ?p=
  await expect(page).toHaveURL(/\?p=/, { timeout: 3_000 });

  // After the cross-fade remount, focus should be restored to the breadcrumb current segment.
  await expect(page.locator('[data-breadcrumb-current]')).toBeFocused({ timeout: 3_000 });

  // Press Escape to return to root level.
  await page.keyboard.press('Escape');
  await expect(page).toHaveURL(/^http:\/\/localhost:3000\/?$/, { timeout: 3_000 });
});

test('keyboard flow: Cmd+K opens query dialog, Escape closes and restores trigger focus', async ({ page }) => {
  await setTheme(page, 'light');
  await page.goto('/');
  await page.waitForLoadState('networkidle');

  // Open via global Cmd+K binding (query-command-dialog.tsx:31).
  await page.keyboard.press('Meta+K');
  const dialog = page.locator('[role="dialog"]').first();
  await expect(dialog).toBeVisible({ timeout: 2_000 });

  // Close with Escape.
  await page.keyboard.press('Escape');
  await expect(dialog).toBeHidden({ timeout: 2_000 });

  // Base UI Dialog default: focus returns to the trigger button. Assert the query trigger has focus.
  await expect(page.getByRole('button', { name: /ask a question/i })).toBeFocused({ timeout: 2_000 });
});
```
