Export interface:
```ts
export interface DataPanelProps {
  title: string;                     // required — flows to SectionHeader.title
  eyebrow?: string;                  // flows to SectionHeader.eyebrow
  description?: string;              // flows to SectionHeader.description
  actions?: React.ReactNode;         // flows to SectionHeader.actions
  children: React.ReactNode;         // required content slot
  footer?: React.ReactNode;          // optional footer slot
  className?: string;                // merged onto outer wrapper
  contentClassName?: string;         // merged onto content wrapper
}
```

Implementation:
```tsx
import type { ReactNode } from 'react';
import { cn } from '@/lib/utils';
import { SectionHeader } from '@/components/layout/section-header';

export function DataPanel({
  title, eyebrow, description, actions, children, footer, className, contentClassName,
}: DataPanelProps) {
  return (
    <div className={cn(
      'rounded-lg bg-surface-raised p-card-padding shadow-elevation-raised',
      className,
    )}>
      <SectionHeader title={title} eyebrow={eyebrow} description={description} actions={actions} />
      <div className={cn('mt-stack', contentClassName)}>{children}</div>
      {footer ? (
        <div className="mt-stack pt-stack border-t border-border">{footer}</div>
      ) : null}
    </div>
  );
}
```

File-top JSDoc block describing: purpose, slot contract (header required, content required, footer optional), composition with SectionHeader, hands-off content sizing policy.
