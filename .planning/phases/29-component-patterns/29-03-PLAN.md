```ts
export type EmptyStateVariant = 'no-data' | 'no-results' | 'error' | 'permissions';

export interface EmptyStateProps {
  variant: EmptyStateVariant;
  title?: string;                     // override default
  description?: string;               // override default
  action?: React.ReactNode | null;    // null to suppress, ReactNode to override default
  onAction?: () => void;              // handler for DEFAULT CTA only (ignored if action prop set)
  className?: string;
}

export function EmptyState(props: EmptyStateProps): JSX.Element;
```
