```ts
import assert from 'node:assert/strict';

// Pure helper extraction ‚Äî copy the logic in isolation:
function maxAgeFromVisible(
  sortedCurves: Array<{ ageInMonths: number }>,
  visibleBatchKeys: string[],
): number {
  const visibleSet = new Set(visibleBatchKeys);
  const visibleCurves = sortedCurves.filter((_, i) => visibleSet.has(`batch_${i}`));
  return Math.max(...visibleCurves.map((c) => c.ageInMonths), 1);
}

// Scenario: 4 batches in sorted order with ages 2, 5, 12, 36.
const curves = [{ ageInMonths: 2 }, { ageInMonths: 5 }, { ageInMonths: 12 }, { ageInMonths: 36 }];

// User shows only the first two (youngest).
assert.equal(maxAgeFromVisible(curves, ['batch_0', 'batch_1']), 5, 'max is 5 when showing 2mo+5mo');

// User shows all ‚Äî max is 36.
assert.equal(maxAgeFromVisible(curves, ['batch_0', 'batch_1', 'batch_2', 'batch_3']), 36);

// User shows only the 12mo batch.
assert.equal(maxAgeFromVisible(curves, ['batch_2']), 12);

// User shows nothing (edge case) ‚Äî falls back to 1.
assert.equal(maxAgeFromVisible(curves, []), 1);

console.log('visible-curves smoke OK');
```
