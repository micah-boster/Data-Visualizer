```ts
/**
 * Coerce a raw BATCH_AGE_IN_MONTHS value into months.
 *
 * Static-cache snapshots from before Phase X normalization may hold days
 * (pre-coercion) instead of months. Values > 365 are treated as days and
 * floored into months; values ≤ 365 are returned as-is.
 *
 * Shared by reshape-curves.ts, the FLT-01 age-bucket filter, and the
 * KPI-01 cascade-tier derivation.
 */
export function coerceAgeMonths(raw: unknown): number {
  const n = Number(raw) || 0;
  return n > 365 ? Math.floor(n / 30) : n;
}
```

```ts
// Simulate the filter predicate used in data-display.filteredRawData:
import { coerceAgeMonths } from '../lib/utils.ts';

function applyAgeFilter<T extends { BATCH_AGE_IN_MONTHS: unknown }>(
  rows: T[],
  age: '3' | '6' | '12' | null,
): T[] {

];
```

```ts
function mergeOverride(
  inferred: { chartType: string; x: string | null; y: string | null } | null,
```
