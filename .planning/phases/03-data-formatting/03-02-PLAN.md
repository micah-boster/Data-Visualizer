Replace the inline `formatAggregate` function with one that delegates to the shared formatters:

```typescript
function formatAggregate(value: number | null, type: string, label: string): string {
  if (value == null) return '\u2014';

  let formatted: string;
  switch (type) {
    case 'currency':
      formatted = formatCurrency(value);
      break;
    case 'percentage':
      formatted = formatPercentage(value);
      break;
    case 'count':
      formatted = formatCount(value);
      break;
    default:
      formatted = formatNumber(value);
      break;
  }
  return `${label}: ${formatted}`;
}
```

This ensures footer aggregates display in the same format as cell values (e.g., footer shows "Sum: $1,234.56" not "Sum: $1,234").
