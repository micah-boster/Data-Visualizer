```typescript
'use client';
import { useSearchParams, usePathname, useRouter } from 'next/navigation';
import { useCallback, useMemo } from 'react';

export type DrillLevel = 'root' | 'partner' | 'batch';
export interface DrillState { level: DrillLevel; partner: string | null; batch: string | null; }

// Param names chosen to avoid collision with use-filter-state.ts's FILTER_PARAMS.
const DRILL_PARTNER_PARAM = 'p';
const DRILL_BATCH_PARAM = 'b';

export function useDrillDown() {
  const searchParams = useSearchParams();
  const pathname = usePathname();
  const router = useRouter();

  // CRITICAL: memo-key off the string, NOT the object. Prevents the
  // "useSearchParams freeze" re-render loop (see RESEARCH.md Pitfall 1).
  const paramsString = searchParams.toString();

  const state: DrillState = useMemo(() => {
    const partner = searchParams.get(DRILL_PARTNER_PARAM);
    const batch = searchParams.get(DRILL_BATCH_PARAM);
    if (partner && batch) return { level: 'batch', partner, batch };
    if (partner) return { level: 'partner', partner, batch: null };
    return { level: 'root', partner: null, batch: null };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [paramsString]);

  const pushWith = useCallback(
    (next: { partner?: string | null; batch?: string | null }) => {
      const params = new URLSearchParams(searchParams.toString());
      if ('partner' in next) {
        if (next.partner) params.set(DRILL_PARTNER_PARAM, next.partner);
        else params.delete(DRILL_PARTNER_PARAM);
      }
      if ('batch' in next) {
        if (next.batch) params.set(DRILL_BATCH_PARAM, next.batch);
        else params.delete(DRILL_BATCH_PARAM);
      }
      const qs = params.toString();
      // NAV-02: push (not replace) so browser back pops a history entry.
      // scroll: false so drill-in doesn't jump user to top (Pitfall 3).
      router.push(qs ? `${pathname}?${qs}` : pathname, { scroll: false });
      // eslint-disable-next-line react-hooks/exhaustive-deps
    },
    [paramsString, pathname, router],
  );

  const drillToPartner = useCallback(
    (partnerName: string) => pushWith({ partner: partnerName, batch: null }),
    [pushWith],
  );

  const drillToBatch = useCallback(
    (batchName: string, partnerName?: string) =>
      pushWith({ partner: partnerName ?? state.partner, batch: batchName }),
    [pushWith, state.partner],
  );

  const navigateToLevel = useCallback(
    (level: DrillLevel) => {
      if (level === 'root') pushWith({ partner: null, batch: null });
      else if (level === 'partner') pushWith({ batch: null });
      // 'batch' is deepest — no-op
    },
    [pushWith],
  );

  return { state, drillToPartner, drillToBatch, navigateToLevel };
}
```

**AVOID:**
- `router.replace()` for drill transitions — breaks NAV-02 back button (use only in Task 2 for stale cleanup if absolutely needed)
- Memoizing off `searchParams` object — use `paramsString`
- `window.history.pushState` / `replaceState` — bypasses Next.js router state
- `encodeURIComponent` on partner names — `URLSearchParams` handles encoding at serialize time
- `nuqs` or new libraries — zero new deps per research
- `'use no memo'` directive — React Compiler opt-outs are the Phase 25 pattern and not needed here; the `paramsString` trick prevents the churn
- Scroll-to-top on push — always pass `{ scroll: false }`

```typescript
// Deep-link stale-param guard (NAV-03). Runs once data is available.
// Non-destructive: we show a toast and navigate to the nearest valid level.
// We use navigateToLevel (router.push) rather than router.replace so the
// user's back button returns them to root — matches CONTEXT's "render empty
// + toast, don't show error page" directive while keeping UX coherent.
useEffect(() => {
  if (isLoading || !data?.data || drillState.level === 'root') return;

  const rows = data.data;
  const partnerExists = drillState.partner
    ? rows.some((r) => getPartnerName(r) === drillState.partner)
    : true;

  if (!partnerExists) {
    toast(`Partner "${drillState.partner}" not found`, {
      description: 'Returning to the root view.',
    });
    navigateToLevel('root');
    return;
  }

  if (drillState.level === 'batch' && drillState.batch) {
    const batchExists = rows.some(
      (r) =>
        getPartnerName(r) === drillState.partner &&
        getBatchName(r) === drillState.batch,
    );
    if (!batchExists) {
      toast(`Batch "${drillState.batch}" not found for ${drillState.partner}`, {
        description: 'Returning to the partner view.',
      });
      navigateToLevel('partner');
    }
  }
}, [isLoading, data?.data, drillState, navigateToLevel]);
```
